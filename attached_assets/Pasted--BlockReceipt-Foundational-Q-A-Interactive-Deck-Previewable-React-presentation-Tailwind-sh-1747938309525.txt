export default function Presentation() {
  const [index, setIndex] = useState(0);

  const prev = () => setIndex((i) => (i > 0 ? i - 1 : slides.length - 1));
  const next = () => setIndex((i) => (i < slides.length - 1 ? i + 1 : 0));

  return (
    <div className="w-screen h-screen bg-gradient-to-br from-emerald-50 to-indigo-100 flex flex-col items-center justify-between p-6 overflow-hidden">
      <AnimatePresence mode="wait" initial={false}>
        <motion.div
          key={index}
          initial={{ opacity: 0, y: 40 }}
          animate={{ opacity: 1, y: 0 }}
          exit={{ opacity: 0, y: -40 }}
          transition={{ duration: 0.4 }}
          className="w-full max-w-5xl"
        >
          <Card className="shadow-2xl rounded-2xl p-8 bg-white/80 backdrop-blur">
            <CardHeader className="text-center space-y-1">
              <h1 className="text-2xl font-bold">{slides[index].title}</h1>
              {slides[index].subtitle && (
                <p className="text-md text-muted-foreground">{slides[index].subtitle}</p>
              )}
            </CardHeader>
            <CardContent>{slides[index].body}</CardContent>
          </Card>
        </motion.div>
      </AnimatePresence>

      <div className="flex items-center gap-4 mt-6">
        <Button variant="outline" size="icon" onClick={prev}>
          <ArrowLeft className="w-5 h-5" />
        </Button>
        <span className="text-sm font-medium">
          {index + 1} / {slides.length}
        </span>
        <Button variant="outline" size="icon" onClick={next}>
          <ArrowRight className="w-5 h-5" />
        </Button>
      </div>
    </div>
  );
}
